001package org.gbif.mybatis.type;
002
003import org.apache.ibatis.type.JdbcType;
004import org.junit.Test;
005import org.junit.runner.RunWith;
006import org.mockito.Mock;
007import org.mockito.runners.MockitoJUnitRunner;
008
009import java.sql.*;
010import java.util.UUID;
011
012import static org.hamcrest.CoreMatchers.*;
013import static org.hamcrest.MatcherAssert.assertThat;
014import static org.junit.Assert.assertNull;
015import static org.mockito.Mockito.verify;
016import static org.mockito.Mockito.when;
017
018@RunWith(MockitoJUnitRunner.class)
019public class UuidTypeHandlerTest {
020
021  private UuidTypeHandler uth = new UuidTypeHandler();
022
023  @Mock
024  private ResultSet mockRs;
025
026  @Mock
027  private CallableStatement mockCs;
028
029  @Mock
030  private PreparedStatement stmt;
031
032  private UUID testUuid = UUID.fromString("b324e8e9-9a4c-44fa-8f1a-7f39ea7ab576");
033
034  @Test
035  public void testSetValidParameter() throws SQLException {
036    uth.setParameter(stmt, 1, testUuid, JdbcType.OTHER);
037    verify(stmt).setObject(1, testUuid.toString(), Types.OTHER);
038  }
039
040  @Test
041  public void testSetNullParameter() throws SQLException {
042    uth.setParameter(stmt, 1, null, JdbcType.OTHER);
043    verify(stmt).setObject(1, null, Types.OTHER);
044  }
045
046  @Test
047  public void testGetInvalidUuid() throws SQLException {
048    when(mockRs.getString("resource_key")).thenReturn("invalid uuid");
049    assertNull(uth.getResult(mockRs, "resource_key"));
050  }
051
052  @Test
053  public void testGetNull() throws SQLException {
054    when(mockRs.getString("resource_key")).thenReturn(null);
055    assertNull(uth.getResult(mockRs, "resource_key"));
056  }
057
058  @Test
059  public void testGetValidUuid() throws SQLException {
060    when(mockRs.getString("resource_key")).thenReturn(testUuid.toString());
061    Object result = uth.getResult(mockRs, "resource_key");
062    assertThat(result, instanceOf(UUID.class));
063    assertThat((UUID) result, equalTo(testUuid));
064  }
065
066  /**
067   * lower case UUID should be the same as lower case UUID.
068   */
069  @Test
070  public void testGetResultLowerCase() throws SQLException {
071    when(mockRs.getString("resource_key")).thenReturn("b324e8e9-9a4c-44fa-8f1a-7f39ea7ab576");
072    assertThat(uth.getResult(mockRs, "resource_key").toString(), is("b324e8e9-9a4c-44fa-8f1a-7f39ea7ab576"));
073
074    when(mockCs.getString(1)).thenReturn("b324e8e9-9a4c-44fa-8f1a-7f39ea7ab576");
075    assertThat(uth.getResult(mockCs, 1).toString(), is("b324e8e9-9a4c-44fa-8f1a-7f39ea7ab576"));
076  }
077
078  /**
079   * Upper case UUID should be converted to lower case and test should be successful.
080   */
081  @Test
082  public void testGetResultUpperCase() throws SQLException {
083    when(mockRs.getString("resource_key")).thenReturn("B324E8E9-9A4C-44FA-8F1A-7F39EA7AB576");
084    assertThat(uth.getResult(mockRs, "resource_key").toString(), is("b324e8e9-9a4c-44fa-8f1a-7f39ea7ab576"));
085
086    when(mockCs.getString(1)).thenReturn("B324E8E9-9A4C-44FA-8F1A-7F39EA7AB576");
087    assertThat(uth.getResult(mockCs, 1).toString(), is("b324e8e9-9a4c-44fa-8f1a-7f39ea7ab576"));
088  }
089
090  /**
091   * Different UUIDs should not be equal.
092   */
093  @Test
094  public void testGetResultNotEqual() throws SQLException {
095    when(mockRs.getString("resource_key")).thenReturn("14c5f0f0-1dab-11d9-8435-b8a03c50a862");
096    assertThat(uth.getResult(mockRs, "resource_key").toString(), not("b324e8e9-9a4c-44fa-8f1a-7f39ea7ab576"));
097
098    when(mockCs.getString(1)).thenReturn("14c5f0f0-1dab-11d9-8435-b8a03c50a862");
099    assertThat(uth.getResult(mockCs, 1).toString(), not("b324e8e9-9a4c-44fa-8f1a-7f39ea7ab576"));
100  }
101}
