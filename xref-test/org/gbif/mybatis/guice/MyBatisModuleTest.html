45    private DataSource provideDs(String dsKey) {
46      Properties  props = new Properties();
47      props.setProperty("dataSourceClassName", "org.h2.jdbcx.JdbcDataSource");
48      props.setProperty("dataSource.user", "sa");
49      props.setProperty("dataSource.password", "sa");
50      props.setProperty("dataSource.url", "jdbc:h2:mem:");
51      props.setProperty("enableCache", "true");
52  
53      MyBatisModule m;
54      if (dsKey != null) {
55        m = new MyBatisModuleImpl(props, dsKey);
56      } else {
57        m = new MyBatisModuleImpl(props);
58      }
59      Injector inj = Guice.createInjector(m);
60      DataSource ds = inj.getInstance(DataSource.class);
61  
62      return ds;
63    }
