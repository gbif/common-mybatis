1   package org.gbif.mybatis.type;
2   
3   import com.google.common.base.Strings;
4   import org.apache.ibatis.type.JdbcType;
5   import org.apache.ibatis.type.TypeHandler;
6   import org.slf4j.Logger;
7   import org.slf4j.LoggerFactory;
8   
9   import java.sql.*;
10  import java.util.UUID;
11  
12  /**
13   * Handler for UUID types.
14   *
15   * @see UUID
16   */
17  public class UuidTypeHandler implements TypeHandler<UUID> {
18      private static final Logger LOG = LoggerFactory.getLogger(UuidTypeHandler.class);
19  
20      @Override
21      public void setParameter(PreparedStatement ps, int i, UUID parameter, JdbcType jdbcType) throws SQLException {
22          if (parameter == null) {
23              ps.setObject(i, null, Types.OTHER);
24          } else {
25              ps.setObject(i, parameter.toString(), Types.OTHER);
26          }
27  
28      }
29  
30      @Override
31      public UUID getResult(ResultSet rs, String columnName) throws SQLException {
32          return toUUID(rs.getString(columnName));
33      }
34  
35      @Override
36      public UUID getResult(ResultSet rs, int columnIndex) throws SQLException {
37          return toUUID(rs.getString(columnIndex));
38      }
39  
40      @Override
41      public UUID getResult(CallableStatement cs, int columnIndex) throws SQLException {
42          return toUUID(cs.getString(columnIndex));
43      }
44  
45      private static UUID toUUID(String val) throws SQLException {
46          if (Strings.isNullOrEmpty(val)) {
47              return null;
48          }
49          try {
50              return UUID.fromString(val);
51          } catch (IllegalArgumentException e) {
52              LOG.warn("Bad UUID found: {}", val);
53          }
54          return null;
55      }
56  
57  }
