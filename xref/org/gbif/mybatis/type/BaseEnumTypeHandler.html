3   import java.sql.CallableStatement;
4   import java.sql.PreparedStatement;
5   import java.sql.ResultSet;
6   import java.sql.SQLException;
7   
8   import org.apache.ibatis.type.JdbcType;
9   import org.apache.ibatis.type.TypeHandler;
10  
11  /**
12   * Base handler for all Enums from the term_type table that are backed by the preferred terms of the term table.
13   *
14   * @param <T> the enumeration to be handled
15   */
16  public class BaseEnumTypeHandler<K, T extends Enum<?>> implements TypeHandler<T> {
17  
18    private final EnumConverter<K, T> typeConverter;
19  
20    public BaseEnumTypeHandler(EnumConverter<K, T> typeConverter) {
21      this.typeConverter = typeConverter;
22    }
23  
24    @Override
25    public void setParameter(PreparedStatement ps, int i, T parameter, JdbcType jdbcType) throws SQLException {
26      ps.setObject(i, typeConverter.fromEnum(parameter));
27    }
28  
29    @Override
30    public T getResult(ResultSet rs, String columnName) throws SQLException {
31      return toEnum((K) rs.getObject(columnName));
32    }
33  
34    @Override
35    public T getResult(ResultSet rs, int columnIndex) throws SQLException {
36      return toEnum((K) rs.getObject(columnIndex));
37    }
38  
39    @Override
40    public T getResult(CallableStatement cs, int columnIndex) throws SQLException {
41      return toEnum((K) cs.getObject(columnIndex));
42    }
43  
44    private T toEnum(K termIndex) {
45      return typeConverter.toEnum(termIndex);
46    }
47  }
