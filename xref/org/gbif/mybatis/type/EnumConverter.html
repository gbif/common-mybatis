3   /**
4    * Generic enumeration value converter interface for mapping magic keys used for persisting
5    * to their respective enum values and vice versa.
6    *
7    * @param <K> the specific key, usually Integer or String this converter is working with
8    * @param <T> the specific enumeration this converter is working with
9    */
10  public interface EnumConverter<K, T extends Enum<?>> {
11  
12    /**
13     * Converts an integer key used for persisting to the matching enumeration value.
14     *
15     * @param key the (term) id used for persisting
16     *
17     * @return the matching enumeration value or NULL if not found
18     */
19    T toEnum(K key);
20  
21    /**
22     * Converts an enumeration value to its unique integer key used for persisting.
23     *
24     * @param value the enumeration value
25     *
26     * @return the matching (term) id used for persisting
27     *
28     * @throws IllegalArgumentException for unknown enumeration values
29     */
30    K fromEnum(T value);
31  }
