1   package org.gbif.mybatis.type;

3   import java.util.Map;
4   
5   import com.google.common.collect.BiMap;
6   import com.google.common.collect.ImmutableBiMap;
7   
8   /**
9    * Generic converter class using an immutable bimap for mapping key/values.
10   *
11   * @param <T> the enumeration to be converted
12   */
13  public abstract class BaseConverter<K, T extends Enum<?>> implements EnumConverter<K, T> {
14  
15    private final BiMap<K, T> map;
16    private final T nullValue;
17  
18    /**
19     * @param map       the map of unique persistency integers to unique enum values.
20     * @param nullValue the value to be used when a key cannot be found.
21     *
22     * @throws IllegalArgumentException if two keys have the same value
23     * @throws NullPointerException     if any key or value is null
24     */
25    protected BaseConverter(T nullValue, Map<K, T> map) {
26      this.map = ImmutableBiMap.copyOf(map);
27      this.nullValue = nullValue;
28    }
29  
30    @Override
31    public T toEnum(K key) {
32      if (key == null) return nullValue;
33      T val = map.get(key);
34      return val == null ? nullValue : val;
35    }
36  
37    @Override
38    public K fromEnum(T value) {
39      if (value == null) return null;
40      if (map.containsValue(value)) {
41        return map.inverse().get(value);
42      }
43      throw new IllegalArgumentException("Enumeration value " + value.name() + " unknown");
44    }
45  }
