3   import org.apache.ibatis.type.BaseTypeHandler;
4   import org.apache.ibatis.type.JdbcType;
5   import org.gbif.common.parsers.UrlParser;
6   
7   import java.net.URI;
8   import java.sql.CallableStatement;
9   import java.sql.PreparedStatement;
10  import java.sql.ResultSet;
11  import java.sql.SQLException;
12  
13  /**
14   * A simple converter for varchars to URI.
15   */
16  public class UriTypeHandler extends BaseTypeHandler<URI> {
17  
18      @Override
19      public void setNonNullParameter(PreparedStatement ps, int i, URI parameter, JdbcType jdbcType) throws SQLException {
20          ps.setString(i, parameter.toString());
21      }
22  
23      @Override
24      public URI getNullableResult(ResultSet rs, String columnName) throws SQLException {
25          return toURI(rs.getString(columnName));
26      }
27  
28      @Override
29      public URI getNullableResult(ResultSet rs, int columnIndex) throws SQLException {
30          return toURI(rs.getString(columnIndex));
31      }
32  
33      @Override
34      public URI getNullableResult(CallableStatement cs, int columnIndex) throws SQLException {
35          return toURI(cs.getString(columnIndex));
36      }
37  
38      private static URI toURI(String val) throws SQLException {
39          return UrlParser.parse(val);
40      }
41  
42  }
