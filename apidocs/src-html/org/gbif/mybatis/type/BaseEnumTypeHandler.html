001package org.gbif.mybatis.type;
002
003import java.sql.CallableStatement;
004import java.sql.PreparedStatement;
005import java.sql.ResultSet;
006import java.sql.SQLException;
007
008import org.apache.ibatis.type.JdbcType;
009import org.apache.ibatis.type.TypeHandler;
010
011/**
012 * Base handler for all Enums from the term_type table that are backed by the preferred terms of the term table.
013 *
014 * @param <T> the enumeration to be handled
015 */
016public class BaseEnumTypeHandler<K, T extends Enum<?>> implements TypeHandler<T> {
017
018  private final EnumConverter<K, T> typeConverter;
019
020  public BaseEnumTypeHandler(EnumConverter<K, T> typeConverter) {
021    this.typeConverter = typeConverter;
022  }
023
024  @Override
025  public void setParameter(PreparedStatement ps, int i, T parameter, JdbcType jdbcType) throws SQLException {
026    ps.setObject(i, typeConverter.fromEnum(parameter));
027  }
028
029  @Override
030  public T getResult(ResultSet rs, String columnName) throws SQLException {
031    return toEnum((K) rs.getObject(columnName));
032  }
033
034  @Override
035  public T getResult(ResultSet rs, int columnIndex) throws SQLException {
036    return toEnum((K) rs.getObject(columnIndex));
037  }
038
039  @Override
040  public T getResult(CallableStatement cs, int columnIndex) throws SQLException {
041    return toEnum((K) cs.getObject(columnIndex));
042  }
043
044  private T toEnum(K termIndex) {
045    return typeConverter.toEnum(termIndex);
046  }
047}
