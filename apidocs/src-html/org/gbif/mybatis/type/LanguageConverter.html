001package org.gbif.mybatis.type;
002
003
004import org.gbif.api.vocabulary.Language;
005
006/**
007 * Enum converter for {@link Language}.
008 */
009public class LanguageConverter implements EnumConverter<String, Language> {
010
011  @Override
012  public String fromEnum(Language value) {
013    return value == null || value == Language.UNKNOWN  ? null : value.getIso2LetterCode();
014  }
015
016  @Override
017  public Language toEnum(String key) {
018    return Language.fromIsoCode(key);
019  }
020
021}
