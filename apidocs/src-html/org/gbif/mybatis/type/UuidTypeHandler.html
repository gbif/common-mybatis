001package org.gbif.mybatis.type;
002
003import com.google.common.base.Strings;
004import org.apache.ibatis.type.JdbcType;
005import org.apache.ibatis.type.TypeHandler;
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009import java.sql.*;
010import java.util.UUID;
011
012/**
013 * Handler for UUID types.
014 *
015 * @see UUID
016 */
017public class UuidTypeHandler implements TypeHandler<UUID> {
018    private static final Logger LOG = LoggerFactory.getLogger(UuidTypeHandler.class);
019
020    @Override
021    public void setParameter(PreparedStatement ps, int i, UUID parameter, JdbcType jdbcType) throws SQLException {
022        if (parameter == null) {
023            ps.setObject(i, null, Types.OTHER);
024        } else {
025            ps.setObject(i, parameter.toString(), Types.OTHER);
026        }
027
028    }
029
030    @Override
031    public UUID getResult(ResultSet rs, String columnName) throws SQLException {
032        return toUUID(rs.getString(columnName));
033    }
034
035    @Override
036    public UUID getResult(ResultSet rs, int columnIndex) throws SQLException {
037        return toUUID(rs.getString(columnIndex));
038    }
039
040    @Override
041    public UUID getResult(CallableStatement cs, int columnIndex) throws SQLException {
042        return toUUID(cs.getString(columnIndex));
043    }
044
045    private static UUID toUUID(String val) throws SQLException {
046        if (Strings.isNullOrEmpty(val)) {
047            return null;
048        }
049        try {
050            return UUID.fromString(val);
051        } catch (IllegalArgumentException e) {
052            LOG.warn("Bad UUID found: {}", val);
053        }
054        return null;
055    }
056
057}
