001package org.gbif.mybatis.type;
002
003import org.apache.ibatis.type.BaseTypeHandler;
004import org.apache.ibatis.type.JdbcType;
005import org.gbif.common.parsers.UrlParser;
006
007import java.net.URI;
008import java.sql.CallableStatement;
009import java.sql.PreparedStatement;
010import java.sql.ResultSet;
011import java.sql.SQLException;
012
013/**
014 * A simple converter for varchars to URI.
015 */
016public class UriTypeHandler extends BaseTypeHandler<URI> {
017
018    @Override
019    public void setNonNullParameter(PreparedStatement ps, int i, URI parameter, JdbcType jdbcType) throws SQLException {
020        ps.setString(i, parameter.toString());
021    }
022
023    @Override
024    public URI getNullableResult(ResultSet rs, String columnName) throws SQLException {
025        return toURI(rs.getString(columnName));
026    }
027
028    @Override
029    public URI getNullableResult(ResultSet rs, int columnIndex) throws SQLException {
030        return toURI(rs.getString(columnIndex));
031    }
032
033    @Override
034    public URI getNullableResult(CallableStatement cs, int columnIndex) throws SQLException {
035        return toURI(cs.getString(columnIndex));
036    }
037
038    private static URI toURI(String val) throws SQLException {
039        return UrlParser.parse(val);
040    }
041
042}
