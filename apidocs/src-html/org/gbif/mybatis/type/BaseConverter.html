001package org.gbif.mybatis.type;
002
003import java.util.Map;
004
005import com.google.common.collect.BiMap;
006import com.google.common.collect.ImmutableBiMap;
007
008/**
009 * Generic converter class using an immutable bimap for mapping key/values.
010 *
011 * @param <T> the enumeration to be converted
012 */
013public abstract class BaseConverter<K, T extends Enum<?>> implements EnumConverter<K, T> {
014
015  private final BiMap<K, T> map;
016  private final T nullValue;
017
018  /**
019   * @param map       the map of unique persistency integers to unique enum values.
020   * @param nullValue the value to be used when a key cannot be found.
021   *
022   * @throws IllegalArgumentException if two keys have the same value
023   * @throws NullPointerException     if any key or value is null
024   */
025  protected BaseConverter(T nullValue, Map<K, T> map) {
026    this.map = ImmutableBiMap.copyOf(map);
027    this.nullValue = nullValue;
028  }
029
030  @Override
031  public T toEnum(K key) {
032    if (key == null) return nullValue;
033    T val = map.get(key);
034    return val == null ? nullValue : val;
035  }
036
037  @Override
038  public K fromEnum(T value) {
039    if (value == null) return null;
040    if (map.containsValue(value)) {
041      return map.inverse().get(value);
042    }
043    throw new IllegalArgumentException("Enumeration value " + value.name() + " unknown");
044  }
045}
